#include <iostream>

#define ll long long

ll fact(ll n)
{
    if (n == 0)
        return 1;
    return n * fact(n - 1);
}

int main()
{
    setlocale(LC_ALL, "Russian");
    std::cout << "Введите целое число: ";
    ll n;
    std::cin >> n;
    std::cout << "Его факториал равен: " << fact(n) << std::endl;
}

#include <iostream>

void rec(int n, int cur = 1)
{
    if (cur == n)
    {
        std::cout << "YES\n";
        exit(0);
    }
    if (cur > n)
        return;
    rec(n, cur + 8);
    rec(n, cur + 9);
}

int main()
{
    int n;
    std::cin >> n;
    rec(n);
    std::cout << "NO\n";
}

#include <iostream>
#include <ctime>

int main()
{
    setlocale(LC_ALL, "Russian");
    srand(time(0));
    int n;
    std::cin >> n;
    int* arr = new int[n];
    //заполняем массив рандомными числами от 0 до 999
    std::cout << "Сгенерированный массив: ";
    for (int i = 0; i < n; ++i)
    {
        arr[i] = rand() % 1000;
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
    int sum = 0;
    for (int i = 0; i < n; ++i)
    {
        if (arr[i] % 7 == 0)
            sum += arr[i];
    }
    delete arr;
    std::cout << "Сумма элементов кратных 7: " << sum << std::endl;
}

#include <iostream>
#include <ctime>

int main()
{
    setlocale(LC_ALL, "Russian");
    srand(time(0));
    int n;
    std::cin >> n;
    int* arr = new int[n];
    //заполняем массив рандомными числами от 0 до 999
    std::cout << "Сгенерированный массив: ";
    for (int i = 0; i < n; ++i)
    {
        arr[i] = rand() % 1000;
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
    int max = -1;
    for (int i = 0; i < n; ++i)
    {
        if (arr[i] % 3 == 0 && max < arr[i])
            max = arr[i];
    }
    delete arr;
    if (max != -1)
        std::cout << "Максимальный элемент кратный 3: " << max << std::endl;
    else
        std::cout << "В последовательности нет чисел кратных 3" << std::endl;
}

#include <iostream>

using namespace std;

int main()
{
    int n, m;
    cin >> n >> m;
    int** matr = new int* [n];
    for (int i = 0; i < n; ++i)
        matr[i] = new int[m];
    for (int i = 0; i < n; ++i)
    {
        for (int j = 0; j < m; ++j)
            cin >> matr[i][j];
    }
    int sum = 0;
    for (int i = 0; i < n; ++i)
    {
        for (int j = 0; j < m; ++j)
        {
            if (matr[i][j] % 7 == 0)
                sum += matr[i][j];
        }
    }
    cout << sum;
}

#include <iostream>
#include <string>

#define SYMB '*'

using namespace std;

int main()
{
    setlocale(LC_ALL, "Russian");
    string s;
    cout << "Введите строку (содержащую только знаки и латинские буквы):" << endl;
    getline(cin, s);
    auto p = s.find(SYMB);
    while (p != string::npos)
    {
        s.erase(p, 1);
        p = s.find(SYMB);
    }
    cout << "Очищенная строка: " << s;
}
